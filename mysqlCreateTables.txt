CREATE DATABASE final;
USE final;
CREATE USER 'user' identified by 'password';
GRANT ALL PRIVILEDGES on final.* to 'user'@'%';

CREATE TABLE `final`.`Users` (`User_ID` INT NOT NULL AUTO_INCREMENT, `email` VARCHAR(50) NOT NULL DEFAULT '', `Name` VARCHAR(50) NOT NULL DEFAULT '', `Password` VARCHAR(50) NOT NULL DEFAULT '', `User_Type` VARCHAR(50) NOT NULL DEFAULT 'Professor', PRIMARY KEY (`User_ID`)) ENGINE = InnoDB;
# user types: 'Professor', 'Admin', 'Super_Admin'


CREATE TABLE `final`.`bookOrder` ( `Order_ID` INT NOT NULL AUTO_INCREMENT , `Title` VARCHAR(50) NOT NULL DEFAULT '', `Deadline` DATETIME DEFAULT NULL, `ISBN` VARCHAR(50) NOT NULL DEFAULT '', `Author` VARCHAR(50) NOT NULL DEFAULT '', `Edition` VARCHAR(50) NOT NULL DEFAULT '', `Publisher` VARCHAR(50) NOT NULL DEFAULT '',  PRIMARY KEY (`Order_ID`)) ENGINE = InnoDB;

CREATE TABLE `final`.`Semester` ( `Sem_ID` INT NOT NULL AUTO_INCREMENT , `Start` DATETIME DEFAULT NULL, `End` DATETIME DEFAULT NULL, PRIMARY KEY (`Sem_ID`)) ENGINE = InnoDB;
