CREATE DATABASE final;
USE final;
CREATE USER 'user' identified by 'password';
GRANT ALL PRIVILEDGES on final.* to 'user'@'%';

CREATE TABLE `Users` (`User_ID` INT NOT NULL AUTO_INCREMENT, `email` VARCHAR(50) NOT NULL DEFAULT '', `Name` VARCHAR(50) NOT NULL DEFAULT '', `Password` VARCHAR(50) NOT NULL DEFAULT '', `User_Type` VARCHAR(50) NOT NULL DEFAULT 'Professor', PRIMARY KEY (`User_ID`)) ENGINE = InnoDB;
# user types: 'Professor', 'Admin', 'Super_Admin'


CREATE TABLE `bookOrder` ( `Order_ID` INT NOT NULL AUTO_INCREMENT , `Title` VARCHAR(50) NOT NULL DEFAULT '',  `ISBN` VARCHAR(50) NOT NULL DEFAULT '', `Author` VARCHAR(50) NOT NULL DEFAULT '', `Edition` VARCHAR(50) NOT NULL DEFAULT '', `Publisher` VARCHAR(50) NOT NULL DEFAULT '', `Req_ID` INT NOT NULL, PRIMARY KEY (`Order_ID`), FOREIGN KEY (`Req_ID`) REFERENCES `RequestForms`(`Req_ID`) ) ENGINE = InnoDB;

CREATE TABLE `Semester` ( `Sem_ID` INT NOT NULL AUTO_INCREMENT , `Season` VARCHAR(50) NOT NULL DEFAULT '', `Year` INT NOT NULL, `Deadline` DATETIME NOT NULL, `Current` BOOLEAN DEFAULT FALSE, PRIMARY KEY (`Sem_ID`)) ENGINE = InnoDB;

CREATE TABLE `RequestForms` (`Req_ID` INT NOT NULL AUTO_INCREMENT, `Sem_ID` INT NOT NULL, `User_ID` INT NOT NULL, FOREIGN KEY (`Sem_ID`) REFERENCES `Semester`(`Sem_ID`), FOREIGN KEY (`User_ID`) REFERENCES `Users`(`User_ID`), PRIMARY KEY (`Req_ID`)) ENGINE = InnoDB;
